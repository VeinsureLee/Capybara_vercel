## 二、Supabase 配置

### 2.1 创建 Supabase 项目

1. 前往 [supabase.com](https://supabase.com) 创建项目
2. 选择离目标用户近的区域（东亚）
3. 记录 `Project URL` 和 `anon key`

### 2.2 配置环境变量

```env
# .env.local
NEXT_PUBLIC_SUPABASE_URL=https://your-project.supabase.co
NEXT_PUBLIC_SUPABASE_ANON_KEY=your-anon-key
SUPABASE_SERVICE_ROLE_KEY=your-service-role-key

DEEPSEEK_API_KEY=your-deepseek-key
CLAUDE_API_KEY=your-claude-key
OPENAI_API_KEY=your-openai-key

CRON_SECRET=your-random-secret-string
```

### 2.3 初始化数据库

在 Supabase SQL Editor 中执行建表 SQL（见设计方案文档的第二章）。

### 2.4 Supabase 客户端封装

**浏览器端客户端：**


**服务端客户端（API Routes）：**

```typescript
// src/lib/supabase/server.ts
import { createServerClient } from '@supabase/ssr'
import { cookies } from 'next/headers'

export async function createServerSupabase() {
  const cookieStore = await cookies()

  return createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() { return cookieStore.getAll() },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value, options }) =>
            cookieStore.set(name, value, options)
          )
        },
      },
    }
  )
}
```

**管理员客户端（Cron Jobs / 后台任务）：**


## 三、AI 客户端实现

### 3.1 多模型 Fallback 客户端

```typescript
// src/lib/ai/client.ts
interface AIProvider {
  name: string
  call: (prompt: string, systemPrompt: string) => Promise<string>
}

const providers: AIProvider[] = [
  {
    name: 'deepseek',
    call: async (prompt, system) => {
      const res = await fetch('https://api.deepseek.com/chat/completions', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${process.env.DEEPSEEK_API_KEY}`,
        },
        body: JSON.stringify({
          model: 'deepseek-chat',
          messages: [
            { role: 'system', content: system },
            { role: 'user', content: prompt },
          ],
          temperature: 0.8,
          max_tokens: 1000,
          response_format: { type: 'json_object' },
        }),
      })
      const data = await res.json()
      return data.choices[0].message.content
    },
  },
  {
    name: 'claude',
    call: async (prompt, system) => {
      const res = await fetch('https://api.anthropic.com/v1/messages', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'x-api-key': process.env.CLAUDE_API_KEY!,
          'anthropic-version': '2023-06-01',
        },
        body: JSON.stringify({
          model: 'claude-haiku-4-5-20251001',
          max_tokens: 1000,
          system,
          messages: [{ role: 'user', content: prompt }],
        }),
      })
      const data = await res.json()
      return data.content[0].text
    },
  },
]

export async function callAI(prompt: string, systemPrompt: string): Promise<string> {
  for (const provider of providers) {
    try {
      const result = await provider.call(prompt, systemPrompt)
      return result
    } catch (error) {
      console.error(`${provider.name} failed:`, error)
      continue
    }
  }
  throw new Error('All AI providers failed')
}
```

## 四、核心 API 实现

### 4.1 对话 API

```typescript
// src/app/api/chat/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { createServerSupabase } from '@/lib/supabase/server'
import { callAI } from '@/lib/ai/client'
import { chatPrompt } from '@/lib/ai/prompts'

export async function POST(req: NextRequest) {
  const supabase = await createServerSupabase()

  // 1. 验证用户
  const { data: { user }, error: authError } = await supabase.auth.getUser()
  if (authError || !user) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  const { message } = await req.json()
  if (!message || typeof message !== 'string') {
    return NextResponse.json({ error: 'Message required' }, { status: 400 })
  }

  // 2. 获取卡皮巴拉信息
  const { data: capybara } = await supabase
    .from('capybaras')
```

```typescript
// src/lib/ai/keywords.ts
import { SupabaseClient } from '@supabase/supabase-js'

export async function aggregateKeywords(
  supabase: SupabaseClient,
  userId: string
): Promise<[string, number][]> {
  const { data: recent } = await supabase
    .from('conversations')
    .select('keywords')
    .eq('user_id', userId)
    .not('keywords', 'is', null)
    .order('created_at', { ascending: false })
    .limit(20)

  if (!recent || recent.length === 0) {
    return [['散步', 1], ['好奇', 0.8]]  // 默认关键词
  }

  const weighted: Record<string, number> = {}
  recent.forEach((conv, i) => {
    const weight = 1 - i * 0.04
    const keywords = conv.keywords as string[]
    keywords?.forEach((kw) => {
      weighted[kw] = (weighted[kw] || 0) + weight
    })
  })

  return Object.entries(weighted)
    .sort(([, a], [, b]) => b - a)
    .slice(0, 10)
}
```

```typescript
      const dayNumber = Math.ceil(
        (now.getTime() - departureDate.getTime()) / (24 * 60 * 60 * 1000)
      )

            .from('memories')
            .select('topic')
            .eq('user_id', travel.matched_user_id)
            .eq('shareable', true)

            name: matchedCapybara.name,
            traits: matchedCapybara.traits,
            sharedTopics: sharedMemories?.map(m => m.topic) || [],
          }
        }

      // 5. 获取前几天的手记摘要
      const { data: prevJournals } = await supabaseAdmin
        .from('journals')

        .eq('travel_id', travel.id)
        .order('day_number', { ascending: true })

        personality: travel.capybaras.traits.join('、'),
        locationName: travel.travel_locations.name,
        locationCountry: travel.travel_locations.country,
        dayNumber,
        totalDays: travel.duration_days,
        encounterCapybara,
        previousJournals: prevJournals?.map(j => j.title),
      })

      let journalContent: any
      try {
        const raw = await callAI(prompt, system)
        journalContent = JSON.parse(raw)
      } catch {
        journalContent = {
          title: `第${dayNumber}天的小记`,
          segments: [
            { type: 'narrative', text: `${travel.capybaras.name}在${travel.travel_locations.name}度过了平静的一天~` }
          ],
        }
      }

      // 7. 保存手记
      await supabaseAdmin.from('journals').insert({
        travel_id: travel.id,
        user_id: travel.user_id,
        day_number: dayNumber,
        title: journalContent.title,
        content: journalContent.segments,
        encounter_capybara_id: encounterCapybara ? undefined : null,
```

### 4.9 Cron: 旅行匹配 (V2)

```typescript
// src/app/api/cron/travel-matching/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { supabaseAdmin } from '@/lib/supabase/admin'
import { calculateMemorySimilarity } from '@/lib/sim/memory-match'

export async function GET(req: NextRequest) {
  const authHeader = req.headers.get('authorization')
  if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  // 1. 获取所有旅行中且未匹配的用户
  const { data: activeTravels } = await supabaseAdmin
    .from('travels')
    .select('id, user_id')
    .eq('status', 'traveling')
    .is('matched_user_id', null)

  if (!activeTravels || activeTravels.length < 2) {
    return NextResponse.json({ matched: 0 })
  }

  // 2. 获取所有旅行用户的 shareable 记忆
  const userMemories: Record<string, any[]> = {}
  for (const travel of activeTravels) {
    const { data: memories } = await supabaseAdmin
      .from('memories')
      .select('topic, emotion, intent')
      .eq('user_id', travel.user_id)
      .eq('shareable', true)
      .order('created_at', { ascending: false })
      .limit(30)

    userMemories[travel.user_id] = memories || []
  }

  // 3. 计算两两相似度，贪心匹配
  const pairs: { a: string; b: string; score: number }[] = []
  const userIds = Object.keys(userMemories)

  for (let i = 0; i < userIds.length; i++) {
    for (let j = i + 1; j < userIds.length; j++) {
      const score = calculateMemorySimilarity(
        userMemories[userIds[i]],
        userMemories[userIds[j]]
      )
      pairs.push({ a: userIds[i], b: userIds[j], score })
    }
  }

  // 按相似度降序排列
  pairs.sort((x, y) => y.score - x.score)

  // 4. 贪心匹配（每个用户最多匹配一次）
  const matched = new Set<string>()
  let matchCount = 0

  for (const pair of pairs) {
    if (matched.has(pair.a) || matched.has(pair.b)) continue
    if (pair.score < 0.3) break  // 相似度阈值

    // 更新双方旅行记录
    const travelA = activeTravels.find(t => t.user_id === pair.a)
    const travelB = activeTravels.find(t => t.user_id === pair.b)

    if (travelA && travelB) {
      await supabaseAdmin
        .from('travels')
        .update({ matched_user_id: pair.b })
        .eq('id', travelA.id)

      await supabaseAdmin
        .from('travels')
        .update({ matched_user_id: pair.a })
        .eq('id', travelB.id)

      matched.add(pair.a)
      matched.add(pair.b)
      matchCount++
    }
  }

  return NextResponse.json({ matched: matchCount })
}
```

### 4.10 记忆匹配算法 (V2)

```typescript
// src/lib/sim/memory-match.ts

interface MemoryVector {
  topic: string
  emotion: string
  intent: string
}

/**
 * 计算两个用户的记忆相似度
 * 公式：0.6 * topic_sim + 0.25 * emotion_sim + 0.15 * intent_sim
 */
export function calculateMemorySimilarity(
  memoriesA: MemoryVector[],
  memoriesB: MemoryVector[]
): number {
  if (!memoriesA.length || !memoriesB.length) return 0

  const topicSim = jaccardSimilarity(
    memoriesA.map(m => m.topic),
    memoriesB.map(m => m.topic)
  )

  const emotionSim = jaccardSimilarity(
    memoriesA.map(m => m.emotion),
    memoriesB.map(m => m.emotion)
  )

  const intentSim = jaccardSimilarity(
    memoriesA.map(m => m.intent),
    memoriesB.map(m => m.intent)
  )

  return 0.6 * topicSim + 0.25 * emotionSim + 0.15 * intentSim
}

function jaccardSimilarity(setA: string[], setB: string[]): number {
  const a = new Set(setA)
  const b = new Set(setB)
  const intersection = new Set([...a].filter(x => b.has(x)))
  const union = new Set([...a, ...b])
  return union.size === 0 ? 0 : intersection.size / union.size
}
```

```typescript
// src/hooks/useRealtime.ts
'use client'

import { useEffect } from 'react'
import { createClient } from '@/lib/supabase/client'
import { useAuth } from '@/components/providers/SupabaseProvider'

export function useExplorationRealtime(
  onMessage: (message: any) => void,
  onComplete: () => void
) {
  const { user } = useAuth()
  const supabase = createClient()

  useEffect(() => {
    if (!user) return

    const channel = supabase
      .channel(`exploration:${user.id}`)
      .on(
        'postgres_changes',
        {
          event: 'INSERT',
          schema: 'public',
          table: 'exploration_messages',
          filter: `user_id=eq.${user.id}`,
        },
        (payload) => onMessage(payload.new)
      )
      .on(
        'postgres_changes',
        {
          event: 'UPDATE',
          schema: 'public',
          table: 'capybaras',
          filter: `owner_id=eq.${user.id}`,
        },
        (payload) => {
          if (payload.new.status === 'home') {
            onComplete()
          }
        }
      )
      .subscribe()

    return () => {
      supabase.removeChannel(channel)
    }
  }, [user])
}
```

```typescript
import { NextResponse, type NextRequest } from 'next/server'

  if (!user && request.nextUrl.pathname.startsWith('/(main)')) {

}
```

## 八、关键注意事项

### Serverless 限制及应对

| 限制 | 影响 | 解决方案 |
|------|------|---------|
| 函数执行时间上限 (60s/Hobby, 300s/Pro) | AI 调用可能超时 | 设置 30s AI 超时 + 降级策略 |
| 无持久内存 | 无法在函数间共享状态 | 所有状态存 Supabase |
| 冷启动延迟 | 首次请求较慢 | 使用 Edge Runtime 或保持活跃 |
| Cron 最小间隔 (Hobby: 每日1次) | 需要更频繁的定时任务 | 升级 Pro Plan ($20/月) |
| 无 WebSocket 支持 | 无法自建长连接 | 使用 Supabase Realtime |

### 数据一致性

- 所有写操作通过 Supabase RLS 保证权限
- 探索完成的奖励发放在 Cron 中用事务处理
- 乐观更新 + 服务端校验

### AI 调用优化

- 对话用 streaming 提升体验
- 探索生成在后台异步完成
- 相同关键词组合缓存探索结果
- AI 不可用时有完整的降级方案
