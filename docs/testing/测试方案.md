## 10.2 单元测试用例

### AI Client (`src/lib/ai/client.ts`)

| 测试用例 | 输入 | 预期 |
|---------|------|------|
| DeepSeek 成功 | 有 API Key + 正常响应 | 返回 AI 文本 |
| DeepSeek 失败降级到 Claude | DeepSeek 超时 | 调用 Claude，返回结果 |
| 全部 Provider 失败 | 无 API Key | 返回 null |
| fallbackChat 关键词匹配 | "好累啊" | 包含治愈回复 + mood: concerned |
| fallbackChat 默认回复 | "随便什么" | 不为空 + mood 合法 |
| fallbackExploration | ["星空"] | 返回合法探索数据 |

```typescript
// 示例测试代码
describe('AI Client', () => {
  test('fallbackChat 匹配"累"关键词', () => {
    const result = fallbackChat('今天好累啊')
    expect(result.mood).toBe('concerned')
    expect(result.reply).toBeTruthy()
    expect(result.want_to_explore).toBe(false)
  })

  test('fallbackExploration 返回合法结构', () => {
    const result = fallbackExploration(['星空', '神秘'])
    expect(result.location).toBeTruthy()
    expect(result.story).toBeTruthy()
    expect(result.items).toBeInstanceOf(Array)
    expect(result.items.length).toBeGreaterThan(0)
    result.items.forEach(item => {
      expect(['common','uncommon','rare','legendary']).toContain(item.rarity)
      expect(['decoration','plant','collectible','interactive']).toContain(item.category)
    })
  })
})
```

### 关键词聚合 (`api/explore`)

| 测试用例 | 输入 | 预期 |
|---------|------|------|
| 空对话列表 | [] | 空数组 |
| 单条对话 | [{keywords:["星空"]}] | ["星空"] |
| 权重衰减 | 最近的"星空"，较早的"大海" | "星空" 排在前面 |
| 去重 | 多条都有"星空" | 只出现一次，权重累加 |
| 上限 8 个 | 20 个不同关键词 | 只返回 top 8 |

```typescript
describe('aggregateKeywords', () => {
  test('时间衰减：最近的权重更高', () => {
    const convs = [
      { keywords: ['星空'], created_at: '2024-01-02' },  // 最近
      { keywords: ['大海'], created_at: '2024-01-01' },  // 较早
    ]
    const result = aggregateKeywords(convs)
    expect(result[0]).toBe('星空')
  })

  test('返回最多 8 个关键词', () => {
    const convs = Array.from({ length: 20 }, (_, i) => ({
      keywords: [`kw${i}`],
      created_at: new Date(Date.now() - i * 60000).toISOString()
    }))
    const result = aggregateKeywords(convs)
    expect(result.length).toBeLessThanOrEqual(8)
  })
})
```

### 进度计算 (`ExploreScene.tsx`)

| 测试用例 | 输入 | 预期 |
|---------|------|------|
| 刚开始 | now = started_at | progress: 0, stage: departure |
| 10% 进度 | 10% 时间过去 | progress: 10, stage: departure |
| 50% 进度 | 一半时间 | progress: 50, stage: journey |
| 80% 进度 | 80% 时间 | progress: 80, stage: discovery |
| 100% 进度 | 时间已过 | progress: 100, stage: return |
| 超过时间 | now > estimated_return | progress: 100 (clamped) |

```typescript
describe('calcProgress', () => {
  const makeExploration = (startOffset: number, duration: number) => ({
    started_at: new Date(Date.now() - startOffset).toISOString(),
    estimated_return: new Date(Date.now() - startOffset + duration).toISOString(),
  })

  test('进度 0% → departure', () => {
    const exp = makeExploration(0, 60000)  // 刚开始，总时长 1 分钟
    const { progress, stage } = calcProgress(exp as any)
    expect(progress).toBeCloseTo(0, 0)
    expect(stage).toBe('departure')
  })

  test('进度 50% → journey', () => {
    const exp = makeExploration(30000, 60000)  // 过了一半
    const { progress, stage } = calcProgress(exp as any)
    expect(progress).toBeCloseTo(50, 0)
    expect(stage).toBe('journey')
  })

  test('进度超过 100% 被钳制', () => {
    const exp = makeExploration(120000, 60000)  // 已超时
    const { progress } = calcProgress(exp as any)
    expect(progress).toBe(100)
  })
})
```

## 10.3 API 集成测试用例

### POST /api/chat

| 测试用例 | 前置条件 | 请求 | 预期响应 |
|---------|---------|------|---------|
| 正常对话 | 已登录 + 有卡皮巴拉 | `{message: "你好"}` | 200 + reply 非空 |
| 未登录 | 无 session | 任意 | 401 Unauthorized |
| 无卡皮巴拉 | 已登录但未创建 | 任意 | 404 或特定错误码 |
| 空消息 | 已登录 | `{message: ""}` | 400 Bad Request |
| 探索中对话 | capybara.status=exploring | 任意 | 200 + 探索相关回复 |

### POST /api/explore

| 测试用例 | 前置条件 | 预期 |
|---------|---------|------|
| 正常出发 | capybara.status=home | 200 + exploration 对象 |
| 重复出发 | 已有 ongoing 探索 | 400 "已在探索中" |
| 非 home 状态 | capybara.status=visiting | 400 "不在家" |
| 未登录 | 无 session | 401 |

### GET /api/explore

| 测试用例 | 前置条件 | 预期 |
|---------|---------|------|
| 无探索 | 无 ongoing 记录 | `{exploration: null}` |
| 进行中 | ongoing + 未到时间 | `{exploration, just_completed: false}` |
| 刚完成 | ongoing + 时间已过 | `{exploration, just_completed: true}` + 状态更新 |

## 10.4 组件测试用例

### ExploreScene

| 测试用例 | 预期 |
|---------|------|
| 进度 < 25% 显示 DepartureStage | DepartureStage 组件已渲染 |
| 进度 25-75% 显示 JourneyStage | JourneyStage 组件已渲染 |

| 测试用例 | 预期 |
|---------|------|
| 物品逐个显示 | 初始 0 个，800ms 后 1 个，之后每 1500ms +1 |
| 稀有度样式 | legendary → 紫色, rare → 蓝色, uncommon → 绿色 |

| 测试用例 | 预期 |

## 10.5 E2E 测试用例

```typescript
// Playwright E2E 测试
describe('核心用户流程', () => {

  test('注册 → 创建卡皮巴拉 → 首次聊天', async ({ page }) => {
    // 注册
    await page.goto('/register')
    await page.fill('[name=email]', 'test@example.com')
    await page.fill('[name=password]', 'test123456')
    await page.fill('[name=nickname]', '测试用户')
    await page.click('button[type=submit]')

    // 应跳转到 /chat
    await page.waitForURL('/chat')

    // 命名卡皮巴拉
    await page.fill('input[placeholder*="名字"]', '小卡')
    await page.click('button:has-text("出发")')

    // 发送消息
    await page.fill('input[placeholder*="说点什么"]', '你好呀')
    await page.click('button:has-text("发送")')

    // 等待回复
    await page.waitForSelector('.capybara-message', { timeout: 30000 })
    const reply = await page.textContent('.capybara-message:last-child')
    expect(reply).toBeTruthy()
  })

  test('探索流程：出发 → 动画 → 完成', async ({ page }) => {
    // 假设已登录
    await page.goto('/explore')

    // 点击出发
    await page.click('button:has-text("出发探索")')

    // 应显示探索动画
    await page.waitForSelector('[data-testid="explore-scene"]')

    // 验证进度条存在
    expect(await page.isVisible('[data-testid="progress-bar"]')).toBe(true)

    // 等待完成（测试中可 mock 时间）
    // ...
  })

  test('底部导航切换', async ({ page }) => {
    await page.goto('/chat')

    // 切换到探索
    await page.click('a[href="/explore"]')
    await page.waitForURL('/explore')

    // 切换到空间
    await page.click('a[href="/home"]')
    await page.waitForURL('/home')

    // 切换回聊天
    await page.click('a[href="/chat"]')
    await page.waitForURL('/chat')
  })
})
```
